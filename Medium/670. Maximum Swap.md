public class Solution {
    public int MaximumSwap(int num) {
        char[] digits = num.ToString().ToArray();
        int[] last = new int[10];
        int n = digits.Length;

        // Step 1: Record the last occurrence of each digit
        for (int i = 0; i < n; i++) {
            last[digits[i] - '0'] = i;
        }

        // Step 2: Greedy scan from left to right
        for (int i = 0; i < n; i++) {
            // Try to find a bigger digit to swap forward
            for (int d = 9; d > digits[i] - '0'; d--) {
                if (last[d] > i) {
                    // Step 3: Swap and return
                    char temp = digits[i];
                    digits[i] = digits[last[d]];
                    digits[last[d]] = temp;
                    return int.Parse(new string(digits));
                }
            }
        }

        // Step 4: No swap needed â€” return original number
        return num;
    }
}
