class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        stack<int> st;
        int maxarea = 0;

        for (int i = 0; i < n; i++) {
            while (!st.empty() && heights[st.top()] >= heights[i]) {
                int k = st.top(); st.pop();
                int prev = st.empty() ? -1 : st.top();
                int area = (i - prev - 1) * heights[k];
                maxarea = max(maxarea, area);
            }
            st.push(i);
        }

        while (!st.empty()) {
            int k = st.top(); st.pop();
            int prev = st.empty() ? -1 : st.top();
            int area = (n - prev - 1) * heights[k];
            maxarea = max(maxarea, area);
        }

        return maxarea;
    }
};
