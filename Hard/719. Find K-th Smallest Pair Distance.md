public class Solution {
    public int SmallestDistancePair(int[] nums, int k) {
        Array.Sort(nums);
        int left = 0;
        int right = nums[^1] - nums[0];

        while (left < right) {
            int mid = (left + right) / 2;
            int count = 0;
            int j = 0;

            for (int i = 0; i < nums.Length; i++) {
                while (j < nums.Length && nums[j] - nums[i] <= mid) {
                    j++;
                }
                count += j - i - 1;
            }

            if (count >= k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }

        return left;
    }
}
